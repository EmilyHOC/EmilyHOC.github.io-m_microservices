# 常用docker命令

# docker 基础命令

systemctl restart docker 重启docker服务

# docker 镜像命令

# 查看所有镜像列表

docker images

# 拉取镜像

docker pull 镜像名

# 给镜像打标记

docker tag 17bdcd448d1e nginx:11.19.4给镜像id为17bdcd448d1e打上nginx:11.19.4标记

# 保存镜像

docker save 镜像名/镜像ID -o 镜像保存在哪个位置与名字

# docker加载打包好的镜像

docker load < tar包 或 docker load -i 镜像保存的位置

# docker 容器命令

# 重启所有容器

docker start $(docker ps -aq)

# 删除所有容器

docker rm -f $(docker ps -aq)

# 查看正在运行的容器

docker ps -a

# 强制删除一个容器

docker rm -f 容器名/容器ID

# 停止容器

docker stop 容器ID/容器名

# 容器文件拷贝

docker cp 容器ID/名称:文件路径 要拷贝到外部的路径

从容器内 拷出 docker cp 容器ID/名称: 容器内路径 容器外路径

从外部 拷贝文件到容器内 docker cp 容器外路径 容器ID/名称: 容器内路径

# 进入容器内部

docker exec -it 容器id bash 退出： exit

# 查看容器的运行日志

docker logs 容器id

# 卷映射

docker run -d -it -p 6500:8080 -v 路径 镜像

-d 后台运行

-v 映射的目录

-p 映射的端口

最后是镜像名
