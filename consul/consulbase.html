# consul

作用： 为Kong提供dns服务

默认启动端口：8500

# 安装consul

sudo yum install -y yum-utils
sudo yum-config-manager --add-repo https://rpm.releases.hashicorp.com/RHEL/hashicorp.repo
sudo yum -y install consul

# consul配置文件

{
    "datacenter": "dc1",
    "data_dir": "/root/consul/data",
    "log_level": "INFO",
    "log_file": "/root/consul/log",
    "log_rotate_bytes": 10485760,
    "log_rotate_max_files": 10,
    "log_json": false,
    "server": true,
    "node_name": "server_main",
    "ui": true,
    "bind_addr": "10.16.193.71",
    "client_addr": "0.0.0.0",
    "bootstrap_expect": 1,
    "ports":{
      "http": 8500,
      "https": 8501,
      "dns": 8600,
      "server": 8300,
      "serf_lan": 8301,
      "serf_wan": 8302
    },
    "auto_encrypt": {
       "allow_tls": false
    }
}

# 启动命令

nohup consul agent --config-file=/root/consul/config/server.json -join main &
