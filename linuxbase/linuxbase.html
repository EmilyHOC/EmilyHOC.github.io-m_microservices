# 常用linux命令

# scp命令（文件拷贝）

scp 本地文件 远程用户名@ip地址:/拷贝的目录

# kill命令（向某个工作传送一个信号）

pkill -9 nomad (杀死nomad 进程)

格式：kill -signal PID

sign数字代表的含义 1 :启动呗终止的进程 2：终端一个程序 9：强制中断 15：正常退出结束 17： 暂停 ）

# pwd 命令（显示当前工作目录）

# ls 命令（列举）

ls 查看目录中的文件

ls -a 包含隐藏文件

ls -l 显示文件和目录的详细资料（例如修改日期）

# ps 命令 （查找）

ps aux （查看系统所有的进程数据）

ps ax （查看不与terminal有关的所有进程）

ps -lA （查看系统所有的进程数据）

ps axjf （查看连同一部分进程树状态）

-e : 显示所有进程 -f : 全格式 -h : 不显示标题 -l : 长格式 -w : 宽输出 a ：显示终端上的所有进程，包括其他用户的进程。 r ：只显示正在运行的进程。 u ：以用户为主的格式来显示程序状况。 x ：显示所有程序，不以终端机来区分。

ps -ef | grep consul 查找consul运行的进程

# chmod命令（改变文件的执行权限）

chmod -R 777 将当前目录及目录下所有文件都给予777权限（所有权限）-R是递归

# rm命令 （删除）

rm -f file1 删除一个叫file1的文件

rmdir dir1 删除一个叫dir1的目录（目录是空的才能删除）

rm -rf dir1 删除一个叫dir1的目录并删除内容

rm -rf dir1 dir2 同时删除两个目录和内容

# mkdir命令（创建目录）

mkdir dir1 创建一个叫做dir1的目录

mkdir dir1 dir2 同时创建两个目录

# mv 命令 （移动）

mv file1 file2 把文件file1重命名为file2

mv file1 file2 dir 把file1,file2移动到dir

# cp 命令（复制）

cp -a file1 file2 连同文件的所有特性把文件file1复制成文件file2

cp dir/* . 复制一个目录下的所有文件到当前工作目录

cp -a /tmp/dir1 . 复制一个目录到当前工作目录

cp -a dir1 dir2 复制一个目录
