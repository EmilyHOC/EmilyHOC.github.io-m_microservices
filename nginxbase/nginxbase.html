# nginx功能

对于前端来讲，nginx可以作为反向代理暴露前端服务

server {

    listen 80;

    server_name localhost;

        location / {

            root    /app;	#root：用于指定访问根目录时，访问虚拟主机的web目录,对于容器来讲是容器内的代码目录
             
             proxy_pass http://localhost:8888;#proxy_pass：反向代理配置
             
            index     index.html index.htm;
    
        }

   }

nginx配置文件的路径/etc/nginx

# 暴露端口

server {

        listen  2022;
        listen  [::]:2022;
        server_name docx;

        location / {
                root  /app;
                index index.html index.htm;
        }
         error_page   500 502 503 504  /50x.html;
                location = /50x.html {
                root   /usr/share/nginx/html;
         }


}
